import pandas as pd
import re
import xlsxwriter

# Cargar los archivos
main_file_path = "1744633215207.csv"
issues_file_path = "issues.csv"

# Leer archivos con codificaci√≥n alternativa si es necesario
encodings = ["utf-8", "utf-8-sig", "latin1"]
for enc in encodings:
    try:
        main_df = pd.read_csv(main_file_path, encoding=enc)
        break
    except Exception:
        continue

for enc in encodings:
    try:
        issues_df = pd.read_csv(issues_file_path, encoding=enc, sep=";")
        break
    except Exception:
        continue

# Extraer n√∫mero RM
def extract_rm_number(text):
    match = re.search(r"RM\s*(\d+)\s*\|", str(text))
    return int(match.group(1)) if match else None

# Extraer etiquetas
def extract_labels(text):
    text = str(text).upper()
    results = {
        "ROBUSTECIMIENTO": "SI" if "ROBUSTECIMIENTO" in text else "NO",
        "CIERRE_CUENTA": "SI" if "CIERRE CUENTA DE INVERSI√ìN" in text or "CIERRE DE BALANCE" in text else "NO",
        "INCIDENCIA_PRODUCTIVA": "SI" if "INCIDENCIA PRODUCTIVA" in text else "NO",
        "PRIORIDAD": None
    }
    prioridad_match = re.findall(r"\[ *PRIORIDAD *\d+ *\]", text)
    estado_prioridad_match = re.findall(r"\b(ALTA|MEDIA|BAJA)\b", text)
    if prioridad_match:
        results["PRIORIDAD"] = prioridad_match[0]
    elif estado_prioridad_match:
        results["PRIORIDAD"] = estado_prioridad_match[0]
    return results

# Aplicar funciones
main_df["N√∫mero RM"] = main_df["Name"].apply(extract_rm_number)
main_df["Estado TRELLO"] = main_df["list"].fillna("")
labels_info = main_df["labels"].apply(extract_labels)
labels_df = pd.DataFrame(labels_info.tolist())
main_processed_df = pd.concat([main_df[["N√∫mero RM", "labels", "Estado TRELLO"]], labels_df], axis=1)

# Unir con issues.csv
issues_df.rename(columns={issues_df.columns[0]: "N√∫mero RM"}, inplace=True)
merged_df = pd.merge(main_processed_df, issues_df, on="N√∫mero RM", how="left")
merged_df["ROBUSTECIMIENTO"] = merged_df["ROBUSTECIMIENTO"].fillna("NO")
merged_df["CIERRE_CUENTA"] = merged_df["CIERRE_CUENTA"].fillna("NO")
merged_df["INCIDENCIA_PRODUCTIVA"] = merged_df["INCIDENCIA_PRODUCTIVA"].fillna("NO")
merged_df["Estado"] = merged_df["Estado"].fillna("")
merged_df["Asunto"] = merged_df["Asunto"].fillna("")
merged_df["Estado RM"] = merged_df["Estado"]
merged_df["Asunto RM"] = merged_df["Asunto"]

final_df = merged_df.rename(columns={
    "N√∫mero RM": "N√∫mero RM",
    "Estado TRELLO": "Estado TRELLO",
    "Estado RM": "Estado RM",
    "Asignado a": "Asignado a",
    "Asunto": "Asunto",
    "ROBUSTECIMIENTO": "ROBUSTECIMIENTO",
    "CIERRE_CUENTA": "CIERRE CUENTA DE INVERSI√ìN",
    "INCIDENCIA_PRODUCTIVA": "INCIDENCIA PRODUCTIVA",
    "PRIORIDAD": "PRIORIDAD"
})[[
    "N√∫mero RM", "Estado TRELLO", "Estado RM", "Asignado a", "Asunto",
    "ROBUSTECIMIENTO", "CIERRE CUENTA DE INVERSI√ìN", "INCIDENCIA PRODUCTIVA",
    "PRIORIDAD"
]]
