import pandas as pd
import re
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment
from openpyxl.utils.dataframe import dataframe_to_rows
from openpyxl.formatting.rule import FormulaRule

# Funciones auxiliares
def extraer_numero_rm(cadena):
    match = re.search(r'RM\s*(\d+)\s*\|', str(cadena))
    return int(match.group(1)) if match else None

def extraer_prioridades(texto):
    prioridades = []
    patrones = [
        r'\[ PRIORIDAD \d+\]',
        r'\[\s*#\d+\s+(ALTA|MEDIA|BAJA)\s*\]',
        r'\b(ALTA|MEDIA|BAJA)\b'
    ]
    for patron in patrones:
        encontrados = re.findall(patron, texto, flags=re.IGNORECASE)
        if encontrados:
            prioridades.extend(encontrados if isinstance(encontrados, list) else [encontrados])
    return ', '.join(set(prioridades)) if prioridades else None

def contiene(texto, cadena):
    return 'SI' if cadena.lower() in texto.lower() else 'NO'

# Lectura segura de archivos
def leer_csv_seguro(path, sep, encoding_list=["utf-8-sig", "latin1"]):
    for enc in encoding_list:
        try:
            return pd.read_csv(path, sep=sep, encoding=enc)
        except Exception:
            continue
    raise ValueError(f"No se pudo leer el archivo: {path}")

# Carga de archivos
archivo_principal = leer_csv_seguro("123456.csv", sep=",")
issues = leer_csv_seguro("issues.csv", sep=";")

# Extracci√≥n y procesamiento
archivo_principal['N√∫mero RM'] = archivo_principal.iloc[:, 0].apply(extraer_numero_rm)
archivo_principal['Estado TRELLO'] = archivo_principal.iloc[:, 3]
archivo_principal['Contenido C'] = archivo_principal.iloc[:, 2].fillna("")
archivo_principal['ROBUSTECIMIENTO'] = archivo_principal['Contenido C'].apply(lambda x: 'SI' if 'ROBUSTECIMIENTO' in x else 'NO')
archivo_principal['CIERRE CUENTA DE INVERSI√ìN'] = archivo_principal['Contenido C'].apply(lambda x: 'SI' if re.search(r'CIERRE CUENTA DE INVERSI√ìN|CIERRE DE BALANCE', x) else 'NO')
archivo_principal['INCIDENCIA PRODUCTIVA'] = archivo_principal['Contenido C'].apply(lambda x: 'SI' if 'INCIDENCIA PRODUCTIVA' in x else 'NO')
archivo_principal['PRIORIDAD'] = archivo_principal['Contenido C'].apply(extraer_prioridades)

# Cruce con issues.csv
issues['N√∫mero RM'] = issues.iloc[:, 0]
archivo_merged = archivo_principal.merge(issues, on='N√∫mero RM', how='left', suffixes=('', '_issues'))

# Completar columnas vac√≠as
archivo_merged['Estado RM'] = archivo_merged['Estado'].fillna("RM CERREDO")
archivo_merged['Asignado a'] = archivo_merged['Asignado a'].fillna("")
archivo_merged['Asunto'] = archivo_merged['Asunto'].fillna("RM CERREDO")

for col in ['ROBUSTECIMIENTO', 'CIERRE CUENTA DE INVERSI√ìN', 'INCIDENCIA PRODUCTIVA']:
    archivo_merged[col] = archivo_merged[col].replace('', 'NO')

# Crear Excel
wb = Workbook()
ws = wb.active
ws.title = "Resumen Consolidado"

columnas_finales = [
    "N√∫mero RM", "Estado TRELLO", "Estado RM", "Asignado a", "Asunto",
    "ROBUSTECIMIENTO", "CIERRE CUENTA DE INVERSI√ìN", "INCIDENCIA PRODUCTIVA", "PRIORIDAD"
]

df_final = archivo_merged[columnas_finales]

# Escribir encabezados
ws.append([col.upper() for col in columnas_finales])
header_fill = PatternFill(start_color="D9D9D9", end_color="D9D9D9", fill_type="solid")
for cell in ws[1]:
    cell.font = Font(bold=True)
    cell.fill = header_fill

# Escribir datos
for r in dataframe_to_rows(df_final, index=False, header=False):
    ws.append(r)

# Centramos columnas F, G, H
for col_letter in ['F', 'G', 'H']:
    for cell in ws[col_letter]:
        cell.alignment = Alignment(horizontal='center')

# Formatos condicionales
amarillo = PatternFill(start_color="FFFF00", end_color="FFFF00", fill_type="solid")
rojo = PatternFill(start_color="FF0000", end_color="FF0000", fill_type="solid")

# Reglas de resaltado condicional
for i in range(2, ws.max_row + 1):
    estado_rm = ws[f'C{i}'].value or ""
    asunto = ws[f'E{i}'].value or ""
    estado_trello = ws[f'B{i}'].value or ""
    asignado = ws[f'D{i}'].value or ""

    if 'Resuelto' in asunto or asunto == "RM CERREDO" or estado_rm == "RM CERREDO":
        for col in range(1, 10):
            ws.cell(row=i, column=col).fill = amarillo

    if estado_trello in [
        "04 En Desarrollo | STAND BY",
        "05 En Desarrollo | EN CURSO",
        "07 En Prueba QA | BACKLOG",
        "08 En Prueba QA | EN CURSO",
        "10 Solicitud de Pasaje a PROD."
    ] and asignado in [
        "Dario Elola", "Nelson Daniel Bord√≥n", "Valentina Juszkiewicz",
        "Renzo Antonioli", "Lucas Gutierrez", "Vanina Zylberg",
        "Carolina Noemi Gualdi", "Franchesca Gabriela Rivas Veliz"
    ]:
        for col in range(1, 10):
            ws.cell(row=i, column=col).fill = rojo

    if estado_trello == "05 En Desarrollo | EN CURSO" and "desarrollo" not in estado_rm.lower():
        for col in range(1, 10):
            ws.cell(row=i, column=col).fill = rojo

    if estado_trello == "04 En Desarrollo | STAND BY" and "stand by" not in estado_rm.lower():
        for col in range(1, 10):
            ws.cell(row=i, column=col).fill = rojo

# Guardar archivo
