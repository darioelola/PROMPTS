import pandas as pd
import re
import os
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment
from openpyxl.utils.dataframe import dataframe_to_rows

# Funci√≥n para leer CSV con m√∫ltiples intentos de codificaci√≥n
def read_csv_with_encoding(file_path, delimiter):
    encodings = ['utf-8', 'utf-8-sig', 'latin1']
    for encoding in encodings:
        try:
            return pd.read_csv(file_path, delimiter=delimiter, encoding=encoding)
        except Exception:
            continue
    raise Exception(f"No se pudo leer el archivo {file_path} con codificaciones comunes.")

# Extracci√≥n de n√∫mero RM de la columna A
def extract_rm_number(text):
    match = re.search(r'RM\s*(\d+)\s*\|', str(text))
    return int(match.group(1)) if match else None

# Extracci√≥n de palabras clave y prioridades desde Columna C
def extract_keywords(text):
    keywords = {
        'ROBUSTECIMIENTO': 'NO',
        'CIERRE CUENTA DE INVERSI√ìN': 'NO',
        'CIERRE DE BALANCE': 'NO',
        'INCIDENCIA PRODUCTIVA': 'NO',
        'PRIORIDAD': ''
    }
    text = str(text).upper()

    if 'ROBUSTECIMIENTO' in text:
        keywords['ROBUSTECIMIENTO'] = 'SI'
    if 'CIERRE CUENTA DE INVERSI√ìN' in text or 'CIERRE DE BALANCE' in text:
        keywords['CIERRE CUENTA DE INVERSI√ìN'] = 'SI'
    if 'INCIDENCIA PRODUCTIVA' in text:
        keywords['INCIDENCIA PRODUCTIVA'] = 'SI'

    prioridad_match = re.findall(r'\[ PRIORIDAD [^\]]+\]|ALTA|MEDIA|BAJA', text)
    if prioridad_match:
        keywords['PRIORIDAD'] = ', '.join(prioridad_match)

    return keywords

# Leer archivos
main_file = [f for f in os.listdir() if f.endswith('.csv') and f != 'issues.csv'][0]
df_main = read_csv_with_encoding(main_file, ',')
df_issues = read_csv_with_encoding('issues.csv', ';')

# Procesamiento de datos
df_main['RM'] = df_main.iloc[:, 0].apply(extract_rm_number)
df_main['Estado TRELLO'] = df_main.iloc[:, 3]
df_main['Texto Analisis'] = df_main.iloc[:, 2].astype(str)

# Extraer palabras clave
keywords_data = df_main['Texto Analisis'].apply(extract_keywords)
keywords_df = pd.DataFrame(keywords_data.tolist())

# Cruce con issues.csv
df_issues['RM'] = df_issues.iloc[:, 0].astype(int)
merged = pd.merge(df_main, df_issues, how='left', left_on='RM', right_on='RM')

# Asunto y relleno de valores por defecto
merged['Asunto'] = merged.get('Asunto', '')
merged['Asunto'].fillna("RM CERREDO", inplace=True)
merged['Estado'].fillna("RM CERREDO", inplace=True)

# Completar columnas de "NO" si est√°n vac√≠as
for col in ['ROBUSTECIMIENTO', 'CIERRE CUENTA DE INVERSI√ìN', 'INCIDENCIA PRODUCTIVA']:
    keywords_df[col] = keywords_df[col].replace('', 'NO')

# Consolidar DataFrame final
final_df = pd.DataFrame({
    'N√∫mero RM': df_main['RM'],
    'Estado TRELLO': df_main['Estado TRELLO'],
    'Estado RM': merged['Estado'],
    'Asignado a': merged['Asignado a'],
    'Asunto': merged['Asunto'],
    'ROBUSTECIMIENTO': keywords_df['ROBUSTECIMIENTO'],
    'CIERRE CUENTA DE INVERSI√ìN': keywords_df['CIERRE CUENTA DE INVERSI√ìN'],
    'INCIDENCIA PRODUCTIVA': keywords_df['INCIDENCIA PRODUCTIVA'],
    'PRIORIDAD': keywords_df['PRIORIDAD'],
})

# Generar Excel
wb = Workbook()
ws = wb.active
ws.title = "Consolidado"

# Escribir encabezados
headers = list(final_df.columns)
ws.append(headers)

# Aplicar formato encabezados
header_fill = PatternFill(start_color="DDDDDD", end_color="DDDDDD", fill_type="solid")
header_font = Font(bold=True)
center_align = Alignment(horizontal="center")

for col_idx, col in enumerate(headers, start=1):
    cell = ws.cell(row=1, column=col_idx)
    cell.font = header_font
    cell.fill = header_fill
    cell.alignment = center_align

# Funciones para aplicar reglas de formato condicional
def apply_formatting(row):
    estado_trello = str(row[1]).strip().upper()
    estado_rm = str(row[2]).strip().upper()
    asignado = str(row[3]).strip().upper()
    asunto = str(row[4]).strip().upper()

    prioridades_trello = [
        "04 EN DESARROLLO | STAND BY",
        "05 EN DESARROLLO | EN CURSO",
        "07 EN PRUEBA QA | BACKLOG",
        "08 EN PRUEBA QA | EN CURSO",
        "10 SOLICITUD DE PASAJE A PROD.",
    ]

    responsables = [
        "DARIO ELOLA", "NELSON DANIEL BORD√ìN", "VALENTINA JUSZKIEWICZ",
        "RENZO ANTONIOLI", "LUCAS GUTIERREZ", "VANINA ZYLBERG",
        "CAROLINA NOEMI GUALDI", "FRANCHESCA GABRIELA RIVAS VELIZ"
    ]

    if asunto == "RM CERREDO" or estado_rm == "RM CERREDO":
        return "YELLOW"
    elif estado_trello == "05 EN DESARROLLO | EN CURSO" and "DESARROLLO" not in estado_rm:
        return "RED"
    elif estado_trello == "04 EN DESARROLLO | STAND BY" and "STAND BY" not in estado_rm:
        return "RED"
    elif estado_trello in prioridades_trello and asignado in responsables:
        return "RED"
    return None

# Agregar datos y aplicar formato
for i, row in final_df.iterrows():
    excel_row = list(row)
    ws.append(excel_row)
    format_type = apply_formatting(excel_row)
    fill_color = None

    if format_type == "YELLOW":
        fill_color = PatternFill(start_color="FFFF00", end_color="FFFF00", fill_type="solid")
    elif format_type == "RED":
        fill_color = PatternFill(start_color="FF0000", end_color="FF0000", fill_type="solid")

    if fill_color:
        for col in range(1, len(headers) + 1):
            ws.cell(row=i + 2, column=col).fill = fill_color

# Centrar columnas F, G, H
for col_letter in ['F', 'G', 'H']:
    for row in ws.iter_rows(min_row=2, max_row=ws.max_row, min_col=ord(col_letter) - 64, max_col=ord(col_letter) - 64):
        for cell in row:
            cell.alignment = center_align

# Guardar archivo Excel
output_file = 'archivo_consolidado.xlsx'
wb.save(output_file)

output_file
